rewrite your docker-compose.yml to have:
- Mautic → MySQL
- Strapi + Prisma → PostgreSQL

…so both containers run side by side without errors.

Here’s a clean, production-ready Docker Compose setup where:
- Mautic runs on MySQL (officially supported)
- Strapi + Prisma run on PostgreSQL
- Volumes and env variables are separated cleanly


x-soloai-volumes:
  &soloai-volumes
  - ${BASE_PATH}/mautic/config:/var/www/html/config:z
  - ${BASE_PATH}/mautic/logs:/var/www/html/var/logs:z
  - ${BASE_PATH}/mautic/media/files:/var/www/html/docroot/media/files:z
  - ${BASE_PATH}/mautic/media/images:/var/www/html/docroot/media/images:z
  - ${BASE_PATH}/cron:/opt/mautic/cron:z
  - mautic-docroot:/var/www/html/docroot:z

services:
  # ────────────── Mautic MySQL Database ──────────────
  mautic-db:
    image: mysql:8.0
    environment:
      - MYSQL_ROOT_PASSWORD=${MYSQL_ROOT_PASSWORD}
      - MYSQL_DATABASE=${MYSQL_DATABASE}
      - MYSQL_USER=${MYSQL_USER}
      - MYSQL_PASSWORD=${MYSQL_PASSWORD}
    ports:
      - "3306:3306"
    volumes: 
      - ${BASE_PATH}/mysql-data:/var/lib/mysql
    healthcheck:
      test: ["CMD-SHELL", "mysqladmin ping -h localhost -u ${MYSQL_USER} -p${MYSQL_PASSWORD}"]
      start_period: 5s
      interval: 5s
      timeout: 5s
      retries: 10
    networks:
      - default

  # ────────────── Strapi PostgreSQL Database ──────────────
  strapi-db:
    image: postgres:16
    environment:
      - POSTGRES_DB=strapi_db
      - POSTGRES_USER=${POSTGRES_STRAPI_USER}
      - POSTGRES_PASSWORD=${POSTGRES_STRAPI_PASSWORD}
    ports:
      - "5432:5432"
    volumes:
      - ${BASE_PATH}/postgres-data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_STRAPI_USER} -d strapi_db"]
      start_period: 5s
      interval: 5s
      timeout: 5s
      retries: 10
    networks:
      - default

  # ────────────── Mautic Web ──────────────
  mautic_web:
    image: mautic/mautic:6-fpm
    links:
      - mautic-db:mysql
    volumes: *soloai-volumes
    environment:
      - DOCKER_MAUTIC_LOAD_TEST_DATA=${DOCKER_MAUTIC_LOAD_TEST_DATA}
      - DOCKER_MAUTIC_RUN_MIGRATIONS=${DOCKER_MAUTIC_RUN_MIGRATIONS}
    env_file:
      - .mautic_env
    healthcheck:
      test: cgi-fcgi -bind -connect 127.0.0.1:9000
      start_period: 5s
      interval: 5s
      timeout: 5s
      retries: 100
    depends_on:
      mautic-db:
        condition: service_healthy
    networks:
      - default

  mautic_cron:
    image: mautic/mautic:6-fpm
    links:
      - mautic-db:mysql
    volumes: *soloai-volumes
    environment:
      - DOCKER_MAUTIC_ROLE=mautic_cron
    env_file:
      - .mautic_env
    depends_on:
      mautic_web:
        condition: service_healthy
    networks:
      - default

  mautic_worker:
    image: mautic/mautic:6-fpm
    links:
      - mautic-db:mysql
    volumes: *soloai-volumes
    environment:
      - DOCKER_MAUTIC_ROLE=mautic_worker
    env_file:
      - .mautic_env
    depends_on:
      mautic_web:
        condition: service_healthy
    networks:
      - default

  # ────────────── Nginx ──────────────
  nginx:
    image: nginx
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf
      - mautic-docroot:/var/www/html/docroot:z
      - ./mautic/media/files:/var/www/html/docroot/media/files:z
      - ./mautic/media/images:/var/www/html/docroot/media/images:z
    depends_on:
      - mautic_web
    ports:
      - 8080:80
    networks:
       - default

  # ────────────── Strapi ──────────────
  strapi:
    image: vshadbolt/strapi
    environment:
      NODE_ENV: development
      DATABASE_CLIENT: postgres
      DATABASE_HOST: ${POSTGRES_HOST}       # should point to strapi-db
      DATABASE_PORT: ${POSTGRES_PORT}       # 5432
      DATABASE_NAME: strapi_db
      DATABASE_USERNAME: ${POSTGRES_STRAPI_USER}
      DATABASE_PASSWORD: ${POSTGRES_STRAPI_PASSWORD}
      JWT_SECRET: ${JWT_SECRET}
      ADMIN_JWT_SECRET: ${ADMIN_JWT_SECRET}
      APP_KEYS: ${APP_KEYS}
      TRANSFER_TOKEN_SALT: ${TRANSFER_TOKEN_SALT}
      API_TOKEN_SALT: ${API_TOKEN_SALT}
      DATABASE_SSL: false
    ports:
      - "1337:1337"
    depends_on:
      strapi-db:
        condition: service_healthy
    volumes:
      - strapi-ap
